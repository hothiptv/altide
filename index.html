// ==================== CODE KẾT NỐI ALTIDE DATA ====================
const GITHUB_USER = "hothiptv";
const REPO_NAME = "altide";

async function layDataTuAltide() {
    // Thêm mốc thời gian ?t= để ép trình duyệt không lưu cache cũ
    const url = `https://raw.githubusercontent.com/\${GITHUB_USER}/\${REPO_NAME}/main/data.json?t=\${Date.now()}`;
    try {
        const response = await fetch(url);
        if(!response.ok) throw new Error("Lỗi kết nối kho lưu trữ!");
        const data = await response.json();
        
        console.log("🎉 Đã kết nối hệ thống dữ liệu thành công!", data);
        
        // Xem toàn bộ danh sách file/mục của ông:
        const files = data.system_files;
        return files; 
        
    } catch (err) {
        console.error("❌ Thất bại: " + err.message);
    }
}
